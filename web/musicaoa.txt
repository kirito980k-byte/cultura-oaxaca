            <h2>Diferencia: Sones y Jarabes</h2>
            <p>La música oaxaqueña se clasifica a menudo en estos dos grandes grupos:</p>
            <ul>
                <li>"Sones:" Tienen un origen más indígena y son predominantes en las regiones de la Sierra y el Istmo. Suelen tener un ritmo marcado que permite el "zapateado" (baile vigoroso), como ocurre en el "Son Istmeño".</li>
                <li>"Jarabes:" Son secuencias de varios sones con un patrón rítmico más rápido y juguetón. Son característicos de las regiones de la Costa y la Mixteca, y se utilizan a menudo en la Danza de la Guelaguetza. Un ejemplo es el "Jarabe Mixteco".</li>
            </ul>

        // Variable para rastrear el elemento de audio que se está reproduciendo actualmente
        let currentAudio = null;
        function toggleAudio(id, button) {
            const audio = document.getElementById(id);
            const isPlaying = !audio.paused;
            
            // Atributo ARIA para accesibilidad
            const ariaLabelBase = button.getAttribute('aria-label').replace('Reproducir ', '').replace('Pausar ', '');
 
            if (isPlaying) {
                audio.pause();
                button.textContent = "Reproducir música";
                button.setAttribute('aria-label', 'Reproducir ' + ariaLabelBase);
                if (currentAudio === audio) {
                    currentAudio = null;
                }
            } else {
                // 1. Pausar el audio actualmente en reproducción (si existe y no es el mismo)
                if (currentAudio && currentAudio !== audio) {
                    currentAudio.pause();
                    const allButtons = document.querySelectorAll('.btn');
                    allButtons.forEach(btn => {
                        if (btn.textContent === "Pausar música") {
                            btn.textContent = "Reproducir música";
                            // Actualizar ARIA de los otros botones
                            const otherAriaBase = btn.getAttribute('aria-label').replace('Pausar ', '').replace('Reproducir ', '');
                            btn.setAttribute('aria-label', 'Reproducir ' + otherAriaBase);
                        }
                    });
                }
                
                audio.play();
                button.textContent = "Pausar música";
                button.setAttribute('aria-label', 'Pausar ' + ariaLabelBase);
                currentAudio = audio;
            }
        }
    </script>
